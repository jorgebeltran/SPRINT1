    private void boton2ActionPerformed(java.awt.event.ActionEvent evt) {                                       
boolean b=IO.textfield_requerido(ID_TAREA);

        if(b){
            
             int codigo = Integer.parseInt(ID_TAREA.getText());
            String tarea=AREA_TEXTO.getText();
            String id=DESIGNACION.getText();
           
          //  String responsable=RESPONSABLE.getSelectedItem().toString();
            String sql="INSERT INTO tarea(ID_TAREA,ID_MIEMBRO,DESCRIPCION)"
                    + " VALUES (?,?,?)";
            try{
                 
                PreparedStatement pst= cn.prepareStatement(sql);
                pst.setInt(1,codigo);
                pst.setString(2,id);
                pst.setString(3,tarea);
            
               
               
                int n=pst.executeUpdate();
                
              if(n>0){
                JOptionPane.showMessageDialog(null, "añadido");
                    } 
            pst.close();
            } catch (Exception e) {
            System.out.println( e.getMessage());
                }
        
        }else{JOptionPane.showMessageDialog(null,"Ingresar Datos");}
            
        limpiar();
             
    }
