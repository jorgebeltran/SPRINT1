    private void btnguardarActionPerformed(java.awt.event.ActionEvent evt) {                                           
        try {
        PreparedStatement pst = cn.prepareStatement("INSERT INTO historiasusuario(yocomo, requiero,talque, complejidad, importancia,condiciones_sa) VALUES (?,?,?,?,?,?)");
        
        pst.setString(1, jcbxYocomo.getSelectedItem().toString());
        pst.setString(2, txRequiero.getText());
        pst.setString(3, texTalque.getText());
        pst.setString(4, txComplejidad.getText());
        pst.setString(5, txImportancia.getText());
        pst.setString(6, txCondSat.getText());
      
        int n = pst.executeUpdate();
        if(n > 0)
        {
            JOptionPane.showMessageDialog(null, "HISTORIA REGISTRADO CON EXITO");
        }else{
            JOptionPane.showMessageDialog(null, "NO SE PUDO GUARDAR REGISTRO");
        }
        
    } catch (Exception e) {
        JOptionPane.showMessageDialog(null, "error"+e);
    }
       
    }
