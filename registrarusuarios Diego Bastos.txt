    private void boton2ActionPerformed(java.awt.event.ActionEvent evt) {                                       
      boolean b=IO.textfield_requerido(NOMBRE,CONTRASE헤);

        if(b){
            
             int codigo = Integer.parseInt(CODIGO.getText());
            String nombres=NOMBRE.getText();
             String correo=CORREO.getText();
              String direccion=DIRECCION_.getText();
            String rol=ROL.getText();
           // String fecha=FECHA.getText();
            String telefono=TELEFONO.getText();
            
            String cargo=CARGO.getSelectedItem().toString();
            String contrase人 = CONTRASE헤.getText();
          //  String responsable=RESPONSABLE.getSelectedItem().toString();
            String sql="INSERT INTO miembro(ID_MIEMBRO,NOMBRE,CORREO,DIRECCION_,ROL,TELEFONO,PASSWORD,TIPOUSUARIO)"
                    + " VALUES (?,?,?,?,?,?,?,?)";
            try{
                PreparedStatement pst= cn.prepareStatement(sql);
                pst.setInt(1,codigo);
                pst.setString(2,nombres);
                pst.setString(3,correo);
            
                pst.setString(4,direccion);
                pst.setString(5, rol);
                pst.setString(6, telefono);
                pst.setString(7,contrase人);
           
                pst.setString(8, cargo);
                int n=pst.executeUpdate();
                
              if(n>0){
                JOptionPane.showMessageDialog(null, "a人dido");
                    } 
            pst.close();
            } catch (Exception e) {
            System.out.println( e.getMessage());
                }
        
        }else{JOptionPane.showMessageDialog(null,"Ingresar Datos");}
            
        limpiar();
    }
